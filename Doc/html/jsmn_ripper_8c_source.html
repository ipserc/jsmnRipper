187 jsmntok_t * findJsmnEngine(list_t * tokenList, char * jsonMsg, jsmntok_t * jsmnTokenArray)
188 {
189  jsmntok_t * jsmnToken = jsmnTokenArray;
190  node_t * tokenListNode = tokenList->head;
191  item_t * item = (item_t *)((node_t *)(tokenListNode->item));
192  char * jsmnTokenItem;
193  int indexCount = 0;
194  int tokenIndex = 0;
195  bool cont; //Continue flag
196 
197  if (getJsmnTokenLen(jsmnToken) < 0) return (jsmntok_t *) NULL;
198  if (!memcmp("", &jsonMsg[jsmnToken->start], (size_t)getJsmnTokenLen(jsmnToken))) return (jsmntok_t *) NULL;
199  cont = nextToken(&jsmnToken);
200  while (cont)
201  {
202  jsmnTokenItem = malloc((size_t)getJsmnTokenLen(jsmnToken)+1);
203  jsmnTokenItem[(size_t)getJsmnTokenLen(jsmnToken)] = '\0';
204  strncpy(jsmnTokenItem, &jsonMsg[jsmnToken->start], (size_t)getJsmnTokenLen(jsmnToken));
205  //TRACE("Token to find item->name:%s", item->name);
206  //TRACE("jsmnTokenItem:%s", jsmnTokenItem);
207  //printf("%s", "***** Test "); printJsmnToken(jsonMsg, jsmnToken); puts("");
208 
209  if (!strcmp(item->name, jsmnTokenItem))
210  {
211  if (item->jtype == JSMN_ARRAY)
212  {
213  tokenIndex = item->index;
214  }
215 
216  cont = nextToken(&jsmnToken);
217  if (jsmnToken->type == JSMN_ARRAY)
218  {
219  cont = nextToken(&jsmnToken);
220  }
221  if (jsmnToken->type == JSMN_OBJECT)
222  {
223  while (tokenIndex > indexCount)
224  {
225  ++indexCount;
226  cont = jumpToTokenPos(&jsmnToken, jsmnToken->end);
227  }
228  cont = nextToken(&jsmnToken);
229  }
230  if (tokenListNode->next)
231  {
232  indexCount = 0;
233  tokenIndex = 0;
234  tokenListNode = tokenListNode->next;
235  item = (item_t *)((node_t *)(tokenListNode->item));
236  //TRACE("**** %s Token to find item->name:%s", (tokenListNode->next) ? "Next" : "Last", item->name);
237  }
238  else // item found
239  {
240  free(jsmnTokenItem);
241  return jsmnToken;
242  }
243  }
244  else
245  {
246  switch (jsmnToken->type)
247  {
248  case JSMN_OBJECT:
249  cont = jumpToTokenPos(&jsmnToken, jsmnToken->end);
250  break;
251  case JSMN_ARRAY:
252  case JSMN_STRING:
253  case JSMN_PRIMITIVE:
254  case JSMN_UNDEFINED:
255  cont = nextToken(&jsmnToken);
256  break;
257  }
258  }
259  free(jsmnTokenItem);
260  }
261  return (jsmntok_t *)NULL;
262 }
