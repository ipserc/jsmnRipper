217 jsmntok_t * findJsmnEngine(list_t * tokenList, char * jsonMsg, jsmntok_t * jsmnTokenArray)
218 {
219  jsmntok_t * jsmnToken = jsmnTokenArray;
220  node_t * tokenListNode = tokenList->head;
221  item_t * item = (item_t *)((node_t *)(tokenListNode->item));
222  char * jsmnTokenItem;
223  int indexCount = 0;
224  int tokenIndex = 0;
225  bool cont; //Continue flag
226 
227  if (getJsmnTokenLen(jsmnToken) < 0) return (jsmntok_t *) NULL;
228  if (!memcmp("", &jsonMsg[jsmnToken->start], (size_t)getJsmnTokenLen(jsmnToken))) return (jsmntok_t *) NULL;
229  cont = nextToken(&jsmnToken);
230  while (cont)
231  {
232  jsmnTokenItem = malloc((size_t)getJsmnTokenLen(jsmnToken)+1);
233  jsmnTokenItem[(size_t)getJsmnTokenLen(jsmnToken)] = '\0';
234  strncpy(jsmnTokenItem, &jsonMsg[jsmnToken->start], (size_t)getJsmnTokenLen(jsmnToken));
235  //TRACE("Token to find item->name:%s", item->name);
236  //TRACE("jsmnTokenItem:%s", jsmnTokenItem);
237  //printf("%s", "***** Test "); printJsmnToken(jsonMsg, jsmnToken); puts("");
238 
239  if (!strcmp(item->name, jsmnTokenItem))
240  {
241  if (item->jtype == JSMN_ARRAY)
242  {
243  tokenIndex = item->index;
244  }
245 
246  cont = nextToken(&jsmnToken);
247  if (jsmnToken->type == JSMN_ARRAY)
248  {
249  cont = nextToken(&jsmnToken);
250  }
251  if (jsmnToken->type == JSMN_OBJECT)
252  {
253  while (tokenIndex > indexCount)
254  {
255  ++indexCount;
256  cont = jumpToTokenPos(&jsmnToken, jsmnToken->end);
257  }
258  cont = nextToken(&jsmnToken);
259  }
260  if (tokenListNode->next)
261  {
262  indexCount = 0;
263  tokenIndex = 0;
264  tokenListNode = tokenListNode->next;
265  item = (item_t *)((node_t *)(tokenListNode->item));
266  //TRACE("**** %s Token to find item->name:%s", (tokenListNode->next) ? "Next" : "Last", item->name);
267  }
268  else // item found
269  {
270  free(jsmnTokenItem);
271  return jsmnToken;
272  }
273  }
274  else
275  {
276  switch (jsmnToken->type)
277  {
278  case JSMN_OBJECT:
279  cont = jumpToTokenPos(&jsmnToken, jsmnToken->end);
280  break;
281  case JSMN_ARRAY:
282  case JSMN_STRING:
283  case JSMN_PRIMITIVE:
284  case JSMN_UNDEFINED:
285  cont = nextToken(&jsmnToken);
286  break;
287  }
288  }
289  free(jsmnTokenItem);
290  }
291  return (jsmntok_t *)NULL;
292 }
